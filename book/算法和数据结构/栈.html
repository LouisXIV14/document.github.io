栈

后进后出的有序集合。新增加或待删除的元素都保存在同一端，新元素都靠近栈顶，旧元素都接近栈底。

function Stack () {
    let items = [];

    // 添加一个或几个元素到栈顶
    this.push = function(ele) {
        items.push(ele);
    }

    // 移除栈顶的元素，并返回被移除的元素
    this.pop = function(ele) {
        if (this.isEmpty()) {
      return undefined;
    }
        items.pop(ele);
    }

    // 返回栈顶的元素，不对栈做任何修改(这个方法不会移除栈顶的元素，仅仅返回它)
    this.peek = function() {
        return items[items.length - 1];
    }

    // 如果栈里没有任何元素就返回 true, 否则返回 false
    this.isEmpty = function() {
        return items.length === 0;
    }

    // 移除栈里的所有元素
    this.clear = function() {
        items = [];
    }

    // 返回栈里的元素个数
    this.size = function() {
        return items.length;
    }

    // 把栈里的元素都输出到控制台
    this.print = function() {
        console.log(items.toString());
    }

}
const stack = new Stack();
stack.push(8, 5);
stack.push(13);
stack.pop();

console.log(stack.peek())
console.log(stack.size())
stack.print()
console.log(stack.isEmpty())

// 基于原型的类比基于函数的类更节省内存
class Stack () {
    constructor() {
        this.items = [];
    }

    // 添加一个或几个元素到栈顶
    push(ele) {
        this.items.push(ele);
    }

    // 移除栈顶的元素，并返回被移除的元素
    pop(ele) {
        if (this.isEmpty()) {
      return undefined;
    }
        return this.items.pop(ele);
    }

    // 返回栈顶的元素，不对栈做任何修改(这个方法不会移除栈顶的元素，仅仅返回它)
    peek() {
        return this.items[items.length - 1];
    }

    // 如果栈里没有任何元素就返回 true, 否则返回 false
    isEmpty() {
        return this.items.length === 0;
    }

    // 移除栈里的所有元素
    clear() {
        this.items = [];
    }

    // 返回栈里的元素个数
    size() {
        return this.items.length;
    }

    // 把栈里的元素都输出到控制台
    print() {
        console.log(this.items.toString());
    }

}

const items = new WeakMap();
class Stack() {
    constructor() {
        items.set(this, []);

    push(ele) {
        let s = items.get(this);
        s.push(ele)
    }

    pop(){
        let s = items.get(this);
        return s.pop;
    }
}

进制转换

// 十进制转换成其他进制
function baseConverter(decNumber, base) {
  // remainder 存储到栈中
  const remStack = new Stack();
  // 十进制转二进制，余数是0～1
  // 十进制转八进制，余数是0～7
  // 十进制转十六进制，余数是0～15(0~9,A~F)
  const digits = '0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ';
  let number = decNumber;
  let rem;
  let baseString = '';

  if (!(base >= 2 && base <= 36)) {
    return '';
  }

  while (number > 0) {
    // JS 不区分是整数还是浮点数，因此需要处理成仅返回整数部分
    rem = Math.floor(number % base);
    remStack.push(rem);
    number = Math.floor(number / base);
  }

  while (!remStack.isEmpty()) {
    baseString += digits[remStack.pop()];
  }

  return baseString;
}
var decNum = 100345, base = 2;
// var decNum = 100345, base = 8;
// var decNum = 100345, base = 16;
console.log(baseConverter(decNum, base));

平衡圆括号

// Stack!
// 平衡圆括号
function parenthesesChecker(symbols) {
  const stack = new Stack();
  const opens = '([{';
  const closers = ')]}';
  let balanced = true;
  let index = 0;
  let symbol;
  let top;

  while (index < symbols.length && balanced) {
    symbol = symbols[index];
    if (opens.indexOf(symbol) >= 0) {
      stack.push(symbol);
    } else if (stack.isEmpty()) {
      balanced = false;
    } else {
      top = stack.pop();
      if (!(opens.indexOf(top) === closers.indexOf(symbol))) {
        balanced = false;
      }
    }
    index++;
  }
  return balanced && stack.isEmpty();
}

console.log('{([])}', parenthesesChecker('{([])}')); // true
console.log('{{([][])}()}', parenthesesChecker('{{([][])}()}')); // true
console.log('[{()]', parenthesesChecker('[{()]')); // false
